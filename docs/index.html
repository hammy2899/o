    o

    common object helper functions which can be used in NodeJS and the browser (supports ES6).

    Installation

    $ npm install --save o

    Usage

    NodeJS

    const o = require('o');
// or
const { /* require specific functions */ is, empty, has, set, get } = require('o');

    Browser (ES6)

    import * as o from 'o';
// or
import { /* require specific functions */ is, empty, has, set, get } from 'o';

    Browser (CDN)

    <script
  type="application/javascript"
  src="https://cdn.jsdelivr.net/npm/o@1.0.0/dist/o.min.js"
/>

    Example usage

    const a = {
  a: 1,
  b: 2,
  c: {
    d: 3,
    e: 4,
  },
};

is(a); // => true
empty(a); // => false
has(a, 'a'); // => true
has(a, 'd'); // => false
has(a, 'c.d'); // => true
a = set(a, 'f', 5);
get(a, 'f'); // => 5
